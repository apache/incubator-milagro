576 /* Method required to calculate x*y+c+r, bottom half in r, top half returned */
577 static inline chunk muladd_384_56(chunk x,chunk y,chunk c,chunk *r)
578 {
579  dchunk prod=(dchunk)x*y+c+*r;
580  *r=(chunk)prod&BMASK_384_56;
581  return (chunk)(prod>>BASEBITS_384_56);
582 }
583 
584 #else
585 
586 /* No integer type available that can store double the wordlength */
587 /* accumulate partial products */
588 
589 static inline chunk muladd_384_56(chunk x,chunk y,chunk c,chunk *r)
590 {
591  chunk x0,x1,y0,y1;
592  chunk bot,top,mid,carry;
593  x0=x&HMASK_384_56;
594  x1=(x>>HBITS_384_56);
595  y0=y&HMASK_384_56;
596  y1=(y>>HBITS_384_56);
597  bot=x0*y0;
598  top=x1*y1;
599  mid=x0*y1+x1*y0;
600  x0=mid&HMASK_384_56;
601  x1=(mid>>HBITS_384_56);
602  bot+=x0<<HBITS_384_56;
603  bot+=*r;
604  bot+=c;
605 
606  top+=x1;
607  carry=bot>>BASEBITS_384_56;
608  bot&=BMASK_384_56;
609  top+=carry;
610 
611  *r=bot;
612  return top;
613 }
