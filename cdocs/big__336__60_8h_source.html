576 /* Method required to calculate x*y+c+r, bottom half in r, top half returned */
577 static inline chunk muladd_336_60(chunk x,chunk y,chunk c,chunk *r)
578 {
579  dchunk prod=(dchunk)x*y+c+*r;
580  *r=(chunk)prod&BMASK_336_60;
581  return (chunk)(prod>>BASEBITS_336_60);
582 }
583 
584 #else
585 
586 /* No integer type available that can store double the wordlength */
587 /* accumulate partial products */
588 
589 static inline chunk muladd_336_60(chunk x,chunk y,chunk c,chunk *r)
590 {
591  chunk x0,x1,y0,y1;
592  chunk bot,top,mid,carry;
593  x0=x&HMASK_336_60;
594  x1=(x>>HBITS_336_60);
595  y0=y&HMASK_336_60;
596  y1=(y>>HBITS_336_60);
597  bot=x0*y0;
598  top=x1*y1;
599  mid=x0*y1+x1*y0;
600  x0=mid&HMASK_336_60;
601  x1=(mid>>HBITS_336_60);
602  bot+=x0<<HBITS_336_60;
603  bot+=*r;
604  bot+=c;
605 
606  top+=x1;
607  carry=bot>>BASEBITS_336_60;
608  bot&=BMASK_336_60;
609  top+=carry;
610 
611  *r=bot;
612  return top;
613 }
